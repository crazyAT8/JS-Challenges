JavaScript Challenges/Exercises No. 2

(Even String's First Half)

        Objective: Write a JavaScript program
                   to extract the first half
                   of a string of even length. 

            Notes: this is going to be tricky                         What's Used: + functions/arrow functions
                   first we examine the string                                     + .length property
                   to see if it's characters                                       + division (/)
                   equal an even number- us modulas (%)                            + .slice() string method 
                   if it is we need to then                                        + if else conditional statement 
                   divide length in half (variable)                                + variables

            Code:
                //1st part - SUCCESS!
                    function stringHalf(str) {
                        let half = str.length/2
                        return half
                    }
                    document.getElementById("result1").innerHTML = stringHalf("worlds");

                //2nd part
                    function stringHalfReturn(str) {
                        let half = str.length/2;
                        let firstHalf = str.slice(0, half);
                        return firstHalf
                    }
                    document.getElementById("result2").innerHTML = stringHalfReturn("worlds");

                //3rd part
                    function evenStr(str) {
                        let x = str.length;
                        if(x % 2 === 0){
                            return "even";
                        } else return "uneven";
                    }
                    document.getElementById("result3").innerHTML = evenStr("worlds");
                    document.getElementById("result4").innerHTML = evenStr("world");

                //4th part
                    function evenStrFirstHalf(str) {
                        let x = str.length;
                        let y = x/2;
                        let a = str.slice(0, y);
                        if(x % 2 === 0) {
                            return a;
                        } else return str;
                    }
                    document.getElementById("result5").innerHTML = evenStrFirstHalf("Missippi");
                    document.getElementById("result6").innerHTML = evenStrFirstHalf("mothers");
                    document.getElementById("result7").innerHTML = evenStrFirstHalf("even");

                //Their version
                    const theirFirstHalf = (str) => str.slice(0, str.length / 2);
        
                    document.getElementById("result8").innerHTML = theirFirstHalf("moms");
                    document.getElementById("result9").innerHTML = theirFirstHalf("mother");
                    document.getElementById("result10").innerHTML = theirFirstHalf("mothers");
